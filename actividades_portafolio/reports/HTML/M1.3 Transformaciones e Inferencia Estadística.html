# Calculando la desviaci√≥n est√°ndar de la resistencia de la botella dado un nivel de confianza del 98%
import scipy.stats as stats
import numpy as np

n = len(df) # Tama√±o de la muestra
s = df.std() # Desviaci√≥n est√°ndar de la muestra
s = s.iloc[0] # Valor de la desviaci√≥n est√°ndar
n_confianza = 0.98 # Nivel de confianza

# Chi-Squared Critical values
chi2_lower = stats.chi2.ppf((1 - n_confianza) / 2, n-1)
chi2_upper = stats.chi2.ppf((1 + n_confianza) / 2, n-1)

lower_var = (n-1) * s**2 / chi2_upper
upper_var = (n-1) * s**2 / chi2_lower

lower_std = np.sqrt(lower_var)
upper_std = np.sqrt(upper_var)

print(f"El intervalo de confianza para la desviaci√≥n est√°ndar de la resistencia de la botella es: ({lower_std.round(4)} - {upper_std.round(4)}) para un nivel de confianza del {n_confianza*100}%")
